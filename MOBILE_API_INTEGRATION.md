# 📱 Mobile App - Backend API Integration Guide\n\nKompletna dokumentacija za povezivanje React Native mobilne aplikacije sa Express backend API-jem.\n\n---\n\n## 🔧 Setup\n\n### 1. Instalacija HTTP klijenta\n\n```bash\ncd mobile\nnpm install axios\n```\n\n### 2. Kreiraj API client\n\n```typescript\n// src/services/api.ts\nimport axios, { AxiosInstance } from 'axios';\nimport AsyncStorage from '@react-native-async-storage/async-storage';\n\nconst API_BASE_URL = process.env.REACT_APP_API_URL || 'http://localhost:3000';\n\nclass APIClient {\n  private client: AxiosInstance;\n  private token: string | null = null;\n\n  constructor() {\n    this.client = axios.create({\n      baseURL: API_BASE_URL,\n      timeout: 10000,\n      headers: {\n        'Content-Type': 'application/json',\n      },\n    });\n\n    // Interceptor - dodaj token u header\n    this.client.interceptors.request.use(\n      async (config) => {\n        const token = await AsyncStorage.getItem('authToken');\n        if (token) {\n          config.headers.Authorization = `Bearer ${token}`;\n        }\n        return config;\n      },\n      (error) => Promise.reject(error)\n    );\n\n    // Interceptor - handle 401 errors\n    this.client.interceptors.response.use(\n      (response) => response,\n      async (error) => {\n        if (error.response?.status === 401) {\n          // Token istekao - logout\n          await AsyncStorage.removeItem('authToken');\n          // Redirect to login\n        }\n        return Promise.reject(error);\n      }\n    );\n  }\n\n  // ============ AUTHENTICATION ============\n\n  async register(email: string, password: string, age: number) {\n    try {\n      const response = await this.client.post('/api/auth/register', {\n        email,\n        password,\n        age,\n      });\n      const { token } = response.data.data;\n      await AsyncStorage.setItem('authToken', token);\n      return response.data.data;\n    } catch (error) {\n      throw this.handleError(error);\n    }\n  }\n\n  async login(email: string, password: string) {\n    try {\n      const response = await this.client.post('/api/auth/login', {\n        email,\n        password,\n      });\n      const { token } = response.data.data;\n      await AsyncStorage.setItem('authToken', token);\n      return response.data.data;\n    } catch (error) {\n      throw this.handleError(error);\n    }\n  }\n\n  async logout() {\n    await AsyncStorage.removeItem('authToken');\n  }\n\n  async registerDevice(deviceToken: string) {\n    try {\n      const response = await this.client.post('/api/auth/register-device', {\n        deviceToken,\n      });\n      return response.data;\n    } catch (error) {\n      throw this.handleError(error);\n    }\n  }\n\n  // ============ PINS ============\n\n  async getPinsByRegion(latitude: number, longitude: number, radius: number) {\n    try {\n      const response = await this.client.get('/api/pins/region', {\n        params: { latitude, longitude, radius },\n      });\n      return response.data.data;\n    } catch (error) {\n      throw this.handleError(error);\n    }\n  }\n\n  async checkPinExists(latitude: number, longitude: number, radiusMeters: number = 100) {\n    try {\n      const response = await this.client.get('/api/pins/check-exists', {\n        params: { latitude, longitude, radiusMeters },\n      });\n      return response.data.data;\n    } catch (error) {\n      throw this.handleError(error);\n    }\n  }\n\n  async createPin(\n    latitude: number,\n    longitude: number,\n    title: string,\n    description: string,\n    actionType: string\n  ) {\n    try {\n      const response = await this.client.post('/api/pins', {\n        latitude,\n        longitude,\n        title,\n        description,\n        actionType,\n      });\n      return response.data.data;\n    } catch (error) {\n      throw this.handleError(error);\n    }\n  }\n\n  async deletePin(pinId: string) {\n    try {\n      const response = await this.client.delete(`/api/pins/${pinId}`);\n      return response.data;\n    } catch (error) {\n      throw this.handleError(error);\n    }\n  }\n\n  // ============ ERROR HANDLING ============\n\n  private handleError(error: any): Error {\n    if (error.response) {\n      const message = error.response.data?.error || error.message;\n      return new Error(message);\n    }\n    return error;\n  }\n}\n\nexport default new APIClient();\n```\n\n---\n\n## 🔐 Authentication Flow\n\n### Register\n\n```typescript\n// src/screens/RegisterScreen.tsx\nimport React, { useState } from src/services/notifications.ts\nimport * as Notifications from 'expo-notifications';\nimport * as Device from 'expo-device';\nimport APIClient from './api';\n\nexport async function setupPushNotifications() {\n  if (!Device.isDevice) {\n    console.log('Push notifications only work on physical devices');\n    return;\n  }\n\n  const { status: existingStatus } = await Notifications.getPermissionsAsync();\n  let finalStatus = existingStatus;\n\n  if (existingStatus !== 'granted') {\n    const { status } = await Notifications.requestPermissionsAsync();\n    finalStatus = status;\n  }\n\n  if (finalStatus !== 'granted') {\n    console.log('Failed to get push token for push notification!');\n    return;\n  }\n\n  const token = (await Notifications.getExpoPushTokenAsync()).data;\n  console.log('Push token:', token);\n\n  // Registruj device token na backend\n  try {\n    await APIClient.registerDevice(token);\n    console.log('Device token registered successfully');\n  } catch (error) {\n    console.error('Failed to register device token:', error);\n  }\n}\n\n// Handle incoming notifications\nexport function setupNotificationHandler() {\n  Notifications.setNotificationHandler({\n    handleNotification: async () => ({\n      shouldShowAlert: true,\n      shouldPlaySound: true,\n      shouldSetBadge: true,\n    }),\n  });\n\n  // Listen for notifications\n  const subscription = Notifications.addNotificationResponseReceivedListener((response) => {\n    const { pinId, latitude, longitude } = response.notification.request.content.data;\n    // Navigate to pin on map\n    console.log('Notification tapped:', pinId);\n  });\n\n  return subscription;\n}\n```\n\n### Initialize u
